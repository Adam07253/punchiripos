📋 Implementation Details:
input.addEventListener("input", function(e) {
  let value = e.target.value;
  // Remove minus sign and non-numeric chars
  value = value.replace(/[^0-9.]/g, '');
  if (e.target.value !== value) {
    e.target.value = value;
  }
});
